import sys
import os
sys.path.append(os.path.dirname(os.path.dirname(os.path.abspath(__file__))))
from core.unclosed_number import UnclosedNumber

def test_not_exist():
    print("=== 测试“不存在”基元 ===\n")
    
    not_exist = UnclosedNumber.not_exist()
    print(f"不存在基元: {not_exist}")
    
    # 连续自乘，每次都对纯缺口操作

    assert one_dim.known_terms == [1]
    assert one_dim.gap_depth == 1
    
    two_dim = not_exist.self_mul().self_mul()   # 第二次自乘：二维
    print(f"二维存在: {two_dim}")
    assert two_dim.anchor == 1

    print(f"三维存在: {three_dim}")
    assert three_dim.anchor == 1
    assert three_dim.known_terms == [396], f"三维应为 [396]，实际为 {three_dim.known_terms}"

· 维度跃迁的正确方式：始终从 NOT_EXIST 开始，连续调用 self_mul() n 次，得到第 n 维的特征整数。
· 不要对一维存在再调用 self_mul()，因为空间型自乘会插入自身，导致嵌套，而不是生成新的维度特征。
· 您的 UnclosedNumber 实现中，anchor=0 的自乘分支已经正确处理了维度生成，每次都会调用 DimensionSeries.generate(current_dim) 返回对应的特征整数（1,30,396…），并生成一个全新的独立未闭环数。
